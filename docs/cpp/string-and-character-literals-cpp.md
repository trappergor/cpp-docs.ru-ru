```cpp
#include <string>
using namespace std::string_literals; // enables s-suffix for std::string literals

int main()
{
    // Character literals
    auto c0 =   'A'; // char
    auto c1 = u8'A'; // char
    auto c2 =  L'A'; // wchar_t
    auto c3 =  u'A'; // char16_t
    auto c4 =  U'A'; // char32_t

    // Multicharacter literals
    auto m0 = 'abcd'; // int, value 0x61626364

    // String literals
    auto s0 =   "hello"; // const char*
    auto s1 = u8"hello"; // const char*, encoded as UTF-8
    auto s2 =  L"hello"; // const wchar_t*
    auto s3 =  u"hello"; // const char16_t*, encoded as UTF-16
    auto s4 =  U"hello"; // const char32_t*, encoded as UTF-32

    // Raw string literals containing unescaped \ and "
    auto R0 =   R"("Hello \ world")"; // const char*
    auto R1 = u8R"("Hello \ world")"; // const char*, encoded as UTF-8
    auto R2 =  LR"("Hello \ world")"; // const wchar_t*
    auto R3 =  uR"("Hello \ world")"; // const char16_t*, encoded as UTF-16
    auto R4 =  UR"("Hello \ world")"; // const char32_t*, encoded as UTF-32

    // Combining string literals with standard s-suffix
    auto S0 =   "hello"s; // std::string
    auto S1 = u8"hello"s; // std::string
    auto S2 =  L"hello"s; // std::wstring
    auto S3 =  u"hello"s; // std::u16string
    auto S4 =  U"hello"s; // std::u32string

    // Combining raw string literals with standard s-suffix
    auto S5 =   R"("Hello \ world")"s; // std::string from a raw const char*
    auto S6 = u8R"("Hello \ world")"s; // std::string from a raw const char*, encoded as UTF-8
    auto S7 =  LR"("Hello \ world")"s; // std::wstring from a raw const wchar_t*
    auto S8 =  uR"("Hello \ world")"s; // std::u16string from a raw const char16_t*, encoded as UTF-16
    auto S9 =  UR"("Hello \ world")"s; // std::u32string from a raw const char32_t*, encoded as UTF-32
}
```

–í —Å–∏–º–≤–æ–ª—å–Ω—ã—Ö –ª–∏—Ç–µ—Ä–∞–ª–∞—Ö –∏ –º–∞—à–∏–Ω–Ω—ã—Ö (–Ω–µ —è–≤–ª—è—é—â–∏—Ö—Å—è –Ω–µ–æ–±—Ä–∞–±–æ—Ç–∞–Ω–Ω—ã–º–∏) —Å—Ç—Ä–æ–∫–æ–≤—ã—Ö –ª–∏—Ç–µ—Ä–∞–ª–∞—Ö –ª—é–±–æ–π —Å–∏–º–≤–æ–ª –º–æ–∂–µ—Ç –±—ã—Ç—å –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω —É–Ω–∏–≤–µ—Ä—Å–∞–ª—å–Ω—ã–º –∏–º–µ–Ω–µ–º —Å–∏–º–≤–æ–ª–∞.  –£–Ω–∏–≤–µ—Ä—Å–∞–ª—å–Ω—ã–µ –∏–º–µ–Ω–∞ —Å–∏–º–≤–æ–ª–æ–≤ —Ñ–æ—Ä–º–∏—Ä—É—é—Ç—Å—è —Å –ø–æ–º–æ—â—å—é –ø—Ä–µ—Ñ–∏–∫—Å–∞ `\U` –∑–∞ –∫–æ—Ç–æ—Ä—ã–º —Å–ª–µ–¥—É–µ—Ç 8-–∑–Ω–∞—á–Ω–∞—è –∫–æ–¥–æ–≤–∞—è —Ç–æ—á–∫–∞ –Æ–Ω–∏–∫–æ–¥–∞ –∏–ª–∏ –ø—Ä–µ—Ñ–∏–∫—Å `\u` –∑–∞ –∫–æ—Ç–æ—Ä—ã–º —Å–ª–µ–¥—É–µ—Ç —á–µ—Ç—ã—Ä–µ—Ö–∑–Ω–∞—á–Ω—ã–π –∫–æ–¥–æ–≤–∞—è —Ç–æ—á–∫–∞ –Æ–Ω–∏–∫–æ–¥–∞. –í—Å–µ –≤–æ—Å–µ–º—å –∏–ª–∏ —á–µ—Ç—ã—Ä–µ –∑–Ω–∞–∫–∞, —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–µ–Ω–Ω–æ, –¥–æ–ª–∂–Ω—ã –ø—Ä–∏—Å—É—Ç—Å—Ç–≤–æ–≤–∞—Ç—å –¥–ª—è —Å–æ–∑–¥–∞–Ω–∏—è –∫–æ—Ä—Ä–µ–∫—Ç–Ω–æ–≥–æ —É–Ω–∏–≤–µ—Ä—Å–∞–ª—å–Ω–æ–≥–æ –∏–º–µ–Ω–∏ —Å–∏–º–≤–æ–ª–∞.

```cpp
char u1 = 'A';          // 'A'
char u2 = '\101';       // octal, 'A'

#### <a name="surrogate-pairs"></a>–°—É—Ä—Ä–æ–≥–∞—Ç–Ω—ã–µ –ø–∞—Ä—ã

–£–Ω–∏–≤–µ—Ä—Å–∞–ª—å–Ω—ã–µ –∏–º–µ–Ω–∞ —Å–∏–º–≤–æ–ª–æ–≤ –Ω–µ –º–æ–≥—É—Ç –∫–æ–¥–∏—Ä–æ–≤–∞—Ç—å –∑–Ω–∞—á–µ–Ω–∏—è –≤ —Å—É—Ä—Ä–æ–≥–∞—Ç–Ω–æ–º –¥–∏–∞–ø–∞–∑–æ–Ω–µ –∫–æ–¥–æ–≤—ã—Ö —Ç–æ—á–µ–∫ D800-DFFF. –î–ª—è —Å—É—Ä—Ä–æ–≥–∞—Ç–Ω—ã—Ö –ø–∞—Ä –Æ–Ω–∏–∫–æ–¥–∞ —É–∫–∞–∂–∏—Ç–µ —É–Ω–∏–≤–µ—Ä—Å–∞–ª—å–Ω–æ–µ –∏–º—è —Å–∏–º–≤–æ–ª–∞, –∏—Å–ø–æ–ª—å–∑—É—è `\UNNNNNNNN`, –≥–¥–µ NNNNNNNN ‚Äî –≤–æ—Å—å–º–∏–∑–Ω–∞—á–Ω–∞—è –∫–æ–¥–æ–≤–∞—è —Ç–æ—á–∫–∞ –¥–ª—è —Å–∏–º–≤–æ–ª–∞. –ü—Ä–∏ –Ω–µ–æ–±—Ö–æ–¥–∏–º–æ—Å—Ç–∏ –∫–æ–º–ø–∏–ª—è—Ç–æ—Ä —Å–æ–∑–¥–∞–µ—Ç —Å—É—Ä—Ä–æ–≥–∞—Ç–Ω—É—é –ø–∞—Ä—É.

–í C++03 —è–∑—ã–∫–æ–º –¥–æ–ø—É—Å–∫–∞–ª–æ—Å—å, —á—Ç–æ–±—ã —É–Ω–∏–≤–µ—Ä—Å–∞–ª—å–Ω—ã–º–∏ –∏–º–µ–Ω–∞–º–∏ —Å–∏–º–≤–æ–ª–æ–≤ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª—è–ª–æ—Å—å –ª–∏—à—å –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–Ω–æ–µ –ø–æ–¥–º–Ω–æ–∂–µ—Å—Ç–≤–æ —Å–∏–º–≤–æ–ª–æ–≤. –¢–∞–∫–∂–µ –º–æ–≥–ª–∏ —Å—É—â–µ—Å—Ç–≤–æ–≤–∞—Ç—å —É–Ω–∏–≤–µ—Ä—Å–∞–ª—å–Ω—ã–µ –∏–º–µ–Ω–∞ —Å–∏–º–≤–æ–ª–æ–≤, –Ω–µ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª—è—é—â–∏–µ –Ω–∏–∫–∞–∫–∏—Ö –¥–æ–ø—É—Å—Ç–∏–º—ã—Ö —Å–∏–º–≤–æ–ª–æ–≤ –Æ–Ω–∏–∫–æ–¥–∞. –≠—Ç–∞ –æ—à–∏–±–∫–∞ –±—ã–ª–∞ –∏—Å–ø—Ä–∞–≤–ª–µ–Ω–∞ –≤ —Å—Ç–∞–Ω–¥–∞—Ä—Ç–µ C++ 11. –í C++11 –≤ —Å–∏–º–≤–æ–ª—å–Ω—ã—Ö –∏ —Å—Ç—Ä–æ–∫–æ–≤—ã—Ö –ª–∏—Ç–µ—Ä–∞–ª–∞—Ö –∏ –∏–¥–µ–Ω—Ç–∏—Ñ–∏–∫–∞—Ç–æ—Ä–∞—Ö –º–æ–∂–Ω–æ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å —É–Ω–∏–≤–µ—Ä—Å–∞–ª—å–Ω—ã–µ –∏–º–µ–Ω–∞ —Å–∏–º–≤–æ–ª–æ–≤.  –î–æ–ø–æ–ª–Ω–∏—Ç–µ–ª—å–Ω—ã–µ —Å–≤–µ–¥–µ–Ω–∏—è –æ–± —É–Ω–∏–≤–µ—Ä—Å–∞–ª—å–Ω—ã—Ö –∏–º–µ–Ω–∞—Ö —Å–∏–º–≤–æ–ª–æ–≤ —Å–º. –≤ —Ä–∞–∑–¥–µ–ª–µ [Character Sets](../cpp/character-sets.md). –î–æ–ø–æ–ª–Ω–∏—Ç–µ–ª—å–Ω—ã–µ —Å–≤–µ–¥–µ–Ω–∏—è –æ –Æ–Ω–∏–∫–æ–¥–µ —Å–º. –≤ —Å—Ç–∞—Ç—å–µ [Unicode](/windows/win32/intl/unicode). –î–æ–ø–æ–ª–Ω–∏—Ç–µ–ª—å–Ω—ã–µ —Å–≤–µ–¥–µ–Ω–∏—è –æ —Å—É—Ä—Ä–æ–≥–∞—Ç–Ω—ã—Ö –ø–∞—Ä–∞—Ö —Å–º. –≤ —Å—Ç–∞—Ç—å–µ [Surrogate Pairs and Supplementary Characters](/windows/win32/Intl/surrogates-and-supplementary-characters)(–°—É—Ä—Ä–æ–≥–∞—Ç–Ω—ã–µ –ø–∞—Ä—ã –∏ –¥–æ–ø–æ–ª–Ω–∏—Ç–µ–ª—å–Ω—ã–µ —Å–∏–º–≤–æ–ª—ã).

## <a name="string-literals"></a>–°—Ç—Ä–æ–∫–æ–≤—ã–µ –ª–∏—Ç–µ—Ä–∞–ª—ã

–°—Ç—Ä–æ–∫–æ–≤—ã–π –ª–∏—Ç–µ—Ä–∞–ª –ø—Ä–µ–¥—Å—Ç–∞–≤–ª—è–µ—Ç –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç—å —Å–∏–º–≤–æ–ª–æ–≤, –∫–æ—Ç–æ—Ä—ã–µ –≤–º–µ—Å—Ç–µ –æ–±—Ä–∞–∑—É—é—Ç —Å—Ç—Ä–æ–∫—É —Å –∑–∞–≤–µ—Ä—à–∞—é—â–∏–º –Ω—É–ª–µ–º. –°–∏–º–≤–æ–ª—ã –¥–æ–ª–∂–Ω—ã –±—ã—Ç—å –∑–∞–∫–ª—é—á–µ–Ω—ã –≤ –¥–≤–æ–π–Ω—ã–µ –∫–∞–≤—ã—á–∫–∏. –°—É—â–µ—Å—Ç–≤—É—é—Ç —Å–ª–µ–¥—É—é—â–∏–µ —Ç–∏–ø—ã —Å—Ç—Ä–æ–∫–æ–≤—ã—Ö –ª–∏—Ç–µ—Ä–∞–ª–æ–≤.

### <a name="narrow-string-literals"></a>–£–∑–∫–∏–µ —Å—Ç—Ä–æ–∫–æ–≤—ã–µ –ª–∏—Ç–µ—Ä–∞–ª—ã

–£–∑–∫–∏–º —Å—Ç—Ä–æ–∫–æ–≤—ã–º –ª–∏—Ç–µ—Ä–∞–ª–æ–º —è–≤–ª—è–µ—Ç—Å—è –Ω–µ—Ñ–∏–∫—Å–∏—Ä–æ–≤–∞–Ω–Ω—ã–π, —Å –¥–≤–æ–π–Ω—ã–º –∫–∞–≤—ã—á–∫–∞–º–∏ –º–∞—Å—Å–∏–≤ —Ç–∏–ø–∞ `const char[n]`, –∑–∞–∫–∞–Ω—á–∏–≤–∞—é—â–∏–π—Å—è –Ω—É–ª–µ–º, –≥–¥–µ n ‚Äî —ç—Ç–æ –¥–ª–∏–Ω–∞ –º–∞—Å—Å–∏–≤–∞ –≤ –±–∞–π—Ç–∞—Ö. –û–±—ã—á–Ω—ã–π —Å—Ç—Ä–æ–∫–æ–≤—ã–π –ª–∏—Ç–µ—Ä–∞–ª –º–æ–∂–µ—Ç —Å–æ–¥–µ—Ä–∂–∞—Ç—å –ª—é–±—ã–µ –≥—Ä–∞—Ñ–∏—á–µ—Å–∫–∏–µ —Å–∏–º–≤–æ–ª—ã, –∑–∞ –∏—Å–∫–ª—é—á–µ–Ω–∏–µ–º –¥–≤–æ–π–Ω—ã—Ö –∫–∞–≤—ã—á–µ–∫ (`"`), –æ–±—Ä–∞—Ç–Ω–æ–π –∫–æ—Å–æ–π —á–µ—Ä—Ç—ã (`\`) –∏–ª–∏ —Å–∏–º–≤–æ–ª–∞ –Ω–æ–≤–æ–π —Å—Ç—Ä–æ–∫–∏. –û–±—ã—á–Ω—ã–π —Å—Ç—Ä–æ–∫–æ–≤—ã–π –ª–∏—Ç–µ—Ä–∞–ª —Ç–∞–∫–∂–µ –º–æ–∂–µ—Ç —Å–æ–¥–µ—Ä–∂–∞—Ç—å –ø–µ—Ä–µ—á–∏—Å–ª–µ–Ω–Ω—ã–µ –≤—ã—à–µ escape-–ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç–∏ –∏ —É–Ω–∏–≤–µ—Ä—Å–∞–ª—å–Ω—ã–µ –∏–º–µ–Ω–∞ —Å–∏–º–≤–æ–ª–æ–≤, –∫–æ—Ç–æ—Ä—ã–µ –ø–æ–º–µ—â–∞—é—Ç—Å—è –≤ –±–∞–π—Ç–µ.

const char *narrow = "abcd";

// represents the string: yes\no

```

#### <a name="utf-8-encoded-strings"></a>–°—Ç—Ä–æ–∫–∏ –≤ –∫–æ–¥–∏—Ä–æ–≤–∫–µ UTF-8

–°—Ç—Ä–æ–∫–∞ –≤ –∫–æ–¥–∏—Ä–æ–≤–∫–µ UTF-8 ‚Äî —ç—Ç–æ U8 —Å –¥–≤–æ–π–Ω–æ–π –∫–∞–≤—ã—á–∫–æ–π, —Ä–∞–∑–¥–µ–ª–µ–Ω–Ω—ã–π –Ω—É–ª–µ–º –º–∞—Å—Å–∏–≤ —Ç–∏–ø–∞ `const char[n]`, –≥–¥–µ *n* ‚Äî —ç—Ç–æ –¥–ª–∏–Ω–∞ –∑–∞–∫–æ–¥–∏—Ä–æ–≤–∞–Ω–Ω–æ–≥–æ –º–∞—Å—Å–∏–≤–∞ –≤ –±–∞–π—Ç–∞—Ö. –°—Ç—Ä–æ–∫–æ–≤—ã–π –ª–∏—Ç–µ—Ä–∞–ª —Å –ø—Ä–µ—Ñ–∏–∫—Å–æ–º u8 –º–æ–∂–µ—Ç —Å–æ–¥–µ—Ä–∂–∞—Ç—å –ª—é–±—ã–µ –≥—Ä–∞—Ñ–∏—á–µ—Å–∫–∏–µ —Å–∏–º–≤–æ–ª—ã, –∑–∞ –∏—Å–∫–ª—é—á–µ–Ω–∏–µ–º –¥–≤–æ–π–Ω—ã—Ö –∫–∞–≤—ã—á–µ–∫ (`"`), –æ–±—Ä–∞—Ç–Ω–æ–π –∫–æ—Å–æ–π —á–µ—Ä—Ç—ã (`\`) –∏–ª–∏ —Å–∏–º–≤–æ–ª–∞ –Ω–æ–≤–æ–π —Å—Ç—Ä–æ–∫–∏. –°—Ç—Ä–æ–∫–æ–≤—ã–π –ª–∏—Ç–µ—Ä–∞–ª —Å –ø—Ä–µ—Ñ–∏–∫—Å–æ–º u8 –º–æ–∂–µ—Ç —Ç–∞–∫–∂–µ —Å–æ–¥–µ—Ä–∂–∞—Ç—å –ø–µ—Ä–µ—á–∏—Å–ª–µ–Ω–Ω—ã–µ –≤—ã—à–µ escape-–ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç–∏ –∏ –ª—é–±—ã–µ —É–Ω–∏–≤–µ—Ä—Å–∞–ª—å–Ω—ã–µ –∏–º–µ–Ω–∞ —Å–∏–º–≤–æ–ª–æ–≤.

### <a name="wide-string-literals"></a>–®–∏—Ä–æ–∫–∏–µ —Å—Ç—Ä–æ–∫–æ–≤—ã–µ –ª–∏—Ç–µ—Ä–∞–ª—ã

–®–∏—Ä–æ–∫–∏–π —Å—Ç—Ä–æ–∫–æ–≤—ã–π –ª–∏—Ç–µ—Ä–∞–ª ‚Äî —ç—Ç–æ –º–∞—Å—Å–∏–≤ –∫–æ–Ω—Å—Ç–∞–Ω—Ç, –∑–∞–∫–∞–Ω—á–∏–≤–∞—é—â–∏–π—Å—è –Ω—É–ª–µ–º, **wchar_t** —Å –ø—Ä–µ—Ñ–∏–∫—Å–æ–º "`L`" –∏ —Å–æ–¥–µ—Ä–∂–∞—â–∏–π –ª—é–±–æ–π –≥—Ä–∞—Ñ–∏—á–µ—Å–∫–∏–π —Å–∏–º–≤–æ–ª, –∫—Ä–æ–º–µ –¥–≤–æ–π–Ω—ã—Ö –∫–∞–≤—ã—á–µ–∫ ("), –æ–±—Ä–∞—Ç–Ω–æ–π –∫–æ—Å–æ–π —á–µ—Ä—Ç—ã (\\) –∏–ª–∏ —Å–∏–º–≤–æ–ª–∞ –Ω–æ–≤–æ–π —Å—Ç—Ä–æ–∫–∏. –†–∞—Å—à–∏—Ä–µ–Ω–Ω—ã–π —Å—Ç—Ä–æ–∫–æ–≤—ã–π –ª–∏—Ç–µ—Ä–∞–ª –º–æ–∂–µ—Ç —Å–æ–¥–µ—Ä–∂–∞—Ç—å –ø–µ—Ä–µ—á–∏—Å–ª–µ–Ω–Ω—ã–µ –≤—ã—à–µ escape-–ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç–∏ –∏ –ª—é–±—ã–µ —É–Ω–∏–≤–µ—Ä—Å–∞–ª—å–Ω—ã–µ –∏–º–µ–Ω–∞ —Å–∏–º–≤–æ–ª–æ–≤.

#### <a name="char16_t-and-char32_t-c11"></a>char16_t –∏ char32_t (C++11)

–í C++11 –¥–æ—Å—Ç—É–ø–Ω—ã —Å–∏–º–≤–æ–ª—å–Ω—ã–µ —Ç–∏–ø—ã `char16_t` (–ø–æ—Ä—Ç–∞—Ç–∏–≤–Ω—ã–π, 16-—Ä–∞–∑—Ä—è–¥–Ω—ã–π –Æ–Ω–∏–∫–æ–¥) –∏ `char32_t` (32-—Ä–∞–∑—Ä—è–¥–Ω—ã–π –Æ–Ω–∏–∫–æ–¥):

### <a name="raw-string-literals-c11"></a>–ù–µ–æ–±—Ä–∞–±–æ—Ç–∞–Ω–Ω—ã–µ —Å—Ç—Ä–æ–∫–æ–≤—ã–µ –ª–∏—Ç–µ—Ä–∞–ª—ã (C++ 11)

–ù–µ–æ–±—Ä–∞–±–æ—Ç–∞–Ω–Ω—ã–π —Å—Ç—Ä–æ–∫–æ–≤—ã–π –ª–∏—Ç–µ—Ä–∞–ª ‚Äî —ç—Ç–æ –º–∞—Å—Å–∏–≤ –∏–∑ –ª—é–±–æ–≥–æ —Å–∏–º–≤–æ–ª—å–Ω–æ–≥–æ —Ç–∏–ø–∞, –∑–∞–∫–∞–Ω—á–∏–≤–∞—é—â–∏–π—Å—è –Ω—É–ª–µ–º, –∫–æ—Ç–æ—Ä—ã–π —Å–æ–¥–µ—Ä–∂–∏—Ç –ª—é–±–æ–π –≥—Ä–∞—Ñ–∏—á–µ—Å–∫–∏–π —Å–∏–º–≤–æ–ª, –≤–∫–ª—é—á–∞—è –¥–≤–æ–π–Ω—ã–µ –∫–∞–≤—ã—á–∫–∏ ("), –æ–±—Ä–∞—Ç–Ω—É—é –∫–æ—Å—É—é —á–µ—Ä—Ç—É (\\) –∏–ª–∏ —Å–∏–º–≤–æ–ª –Ω–æ–≤–æ–π —Å—Ç—Ä–æ–∫–∏. –ù–µ–æ–±—Ä–∞–±–æ—Ç–∞–Ω–Ω—ã–µ —Å—Ç—Ä–æ–∫–æ–≤—ã–µ –ª–∏—Ç–µ—Ä–∞–ª—ã —á–∞—Å—Ç–æ –ø—Ä–∏–º–µ–Ω—è—é—Ç—Å—è –≤ —Ä–µ–≥—É–ª—è—Ä–Ω—ã—Ö –≤—ã—Ä–∞–∂–µ–Ω–∏—è—Ö, –∫–æ—Ç–æ—Ä—ã–µ –∏—Å–ø–æ–ª—å–∑—É—é—Ç –∫–ª–∞—Å—Å—ã —Å–∏–º–≤–æ–ª–æ–≤, –∞ —Ç–∞–∫–∂–µ –≤ —Å—Ç—Ä–æ–∫–∞—Ö HTML –∏ XML. –ü—Ä–∏–º–µ—Ä—ã —Å–º. –≤ —Å–ª–µ–¥—É—é—â–µ–π —Å—Ç–∞—Ç—å–µ: [Bjarne Stroustrup's FAQ on C++11](http://www.stroustrup.com/C++11FAQ.html)(–í–æ–ø—Ä–æ—Å—ã –∏ –æ—Ç–≤–µ—Ç—ã –æ C++11 –æ—Ç –ë—å–µ—Ä–Ω–∞ –°—Ç—Ä–∞—É—Å—Ç—Ä—É–ø–∞).
